## COMMs構築(Gitlab, Mattermost)
- docker-compose.yml
```yml
version: "3"
services:
  web:
    image: gitlab/gitlab-ee:latest
    volumes:
      - "gitlab_config:/etc/gitlab"
      - "gitlab_logs:/var/log/gitlab"
      - "gitlab_data:/var/opt/gitlab"
    restart: always
    ports:
      - "3100:3100"
      - "8001:8001"
    hostname: '192.168.10.70'
    environment:
      GITLAB_OMNIBUS_CONFIG: |
        external_url 'http://192.168.10.70:3100'
        mattermost_external_url 'http://192.168.10.70:8001'
    shm_size: '256m'
volumes:
  gitlab_config: {}
  gitlab_logs: {}
  gitlab_data: {}
```

- 補足
  - root初期パスワード
    - コンテナ内の`/etc/gitlab/initial_root_password`に記載
    - Gitlab URL: `external_url`の値
    - Mattermost URL: `mattermost_external_url`の値


## Gitlab - Drawio 連携（Repository保存）
### Gitlab側の設定
1. 管理者ログイン
2. `Admin area` -> `Applications`
  - `http://[Gitlab Host]/admin/applications` 
3. "Add new application"
  - Name: 任意 
  - Redirect URI: `http://[drawio Host]/gitlab`
  - Scopes: `api`, `read_repository`, `write_repository`
4. 作成すると`Application ID`と`Secret`が発行されるので記録

### Drawio側の設定(`docker-compose.yml`)
```
version: "3"
services:
  drawio:
    image: jgraph/drawio:latest
    ports:
      - "3300:8080"
    environment:
      - DRAWIO_GITLAB_URL=http://[Gitlab Host]
      - DRAWIO_GITLAB_ID=[Application ID]
      - DRAWIO_GITLAB_SECRET=[Secret]
      - DRAWIO_SERVER_URL=http://[Drawio Host]/
      - DRAWIO_USE_HTTP=1
    command: bash -c "echo 'window.ALLOW_CUSTOM_PLUGINS = true' >> /usr/local/tomcat/webapps/draw/js/PreConfig.js && sed -i \"s/script-src \\\\\\'self\\\\\\'/script-src \\\\\\'self\\\\\\' */\" /usr/local/tomcat/webapps/draw/js/PreConfig.js  && catalina.sh run"
    restart: always
```


## Gitlab - Mattermost Webhook 設定 （BOT設定）
- Webhook設定することでGitlabの以下の操作が行われた場合にMattermostにて通知を受け取ることが可能
  - issueの作成
  - コメントの挿入
  - wikiの作成
  - repositoryへのpush
    - 共有ファイルの保存
    - NW図等のdrawioの作成または更新（GitlabとDrawioを連携していた場合）
- LeaderやTech-Leaderが当該Mattermostのチャンネルを監視することで、チームでの作業状況について早期に把握可能

### Mattermost側の設定
1. 管理者ログイン
2. 通知を受け取るチャンネル作成
3. 左上のアイコンをクリック -> Integrations
4. Incomming Webhooks
5. Add Incoming Webhook
  - Title: 任意
  - Description: 任意
  - Channel: 作成したチャンネルを選択
6. WebhookのURLが発行されるので記録

### Gitlab側の設定
#### 初回のみ
1. 管理者ログイン
2. Admin area -> Settings -> Network
  - `http://[Giglab Host]/admin/application_settings/network`
3. Outbound requests
4. "Allow requests to the local network from webhooks and integrations"にチェック
5. Save changes

#### プロジェクト単位
1. Mattermostに通知を出したいプロジェクトを選択
2. Settings -> Integrations
3. Mattermost notifications
  - Trigger: 任意（以下を推奨）
    - `A push is made to the repository`
    - `An issue is created, closed, or reopened`
    - `A comment is added`
    - `A wiki page is created or updated`
  - Webhook: MattermostでWebhookを作成した際に発行された`Webhook URL`を入力


## ネットワーク図一括描画(Drawio)
- 本記載の内容は最低限のコードサンプルであり、そのまま運用で使えるものではない
- ネットワーク図作成においての要領（使用ツール、順番等）や描画イメージ（レイアウト、ホストの単位（IP単位or物理的なホスト単位）等）の確立後、当該要領に合わせてコードをチューニングする必要がある
- ３つのやり方を紹介するがそれぞれに特性があるため、要領にマッチしたやり方を選択しチューニングすることを推奨する

    function insert_nmap(result){
        var parent = graph.getDefaultParent();

        const parser = new DOMParser();
        var xml = parser.parseFromString(result, 'text/xml');
        
        var styles = {
            'ws': 'verticalLabelPosition=bottom;sketch=0;aspect=fixed;html=1;verticalAlign=top;strokeColor=none;align=center;outlineConnect=0;shape=mxgraph.citrix.desktop;',
            'network': '"sketch=0;html=1;aspect=fixed;strokeColor=none;shadow=0;align=center;verticalAlign=middle;fillColor=lightblue;shape=mxgraph.gcp2.network',
        }
        var dataset = [];
        var networks = [];
        var objects = {};
        var current_objects = {};

        var hosts = xml.querySelectorAll('host');

        for(host of hosts){
            var data = {};
            
            var addresses = host.querySelectorAll('address');
            for(address of addresses){
                if(address.getAttribute('addrtype') == 'ipv4'){
                    data['address'] = address.getAttribute('addr');
                }
            }
            
            data['port'] = [];
            var ports = host.querySelectorAll('port');
            for(port of ports){
                var state = port.querySelector('state');
                if(state.getAttribute('state') == 'open'){
                    data['port'].push(port.getAttribute('portid') + '/' + port.getAttribute('protocol').toUpperCase());
                }
            }

            data['type'] = 'ws';
				
            var address_split = data.address.split('.');
            var network  = address_split[0] + '.' + address_split[1] + '.' + address_split[2] + '.' + '0/24';

            data['network'] = network;

            if(!networks.includes(network)){
                dataset.push({
                    'address': network,
                    'type': 'network',
                });
                networks.push(network);
            }
            
            dataset.push(data);
        }

        var cells = graph.model.cells;
        for(id in cells){
            var cell = cells[id];
            if(cell.getAttribute('IPAddress')){
                current_objects[cell.getAttribute('IPAddress')] = cell;
            }
        }

        for(data of dataset){
            if(current_objects[data.address]){
                var node = current_objects[data.address];
            }else{
                var doc = mxUtils.createXmlDocument();
                var node = doc.createElement('MyNode');
            }
            node.setAttribute('label', '%IPAddress%');
            node.setAttribute('IPAddress', data.address);
            node.setAttribute('type', data.type);
            node.setAttribute('placeholders', 1);
            if(data.port){
                node.setAttribute('Port', data.port.join(', '));
            }
            if(data.network){
                node.setAttribute('Network', data.network);
            }
            if(current_objects[data.address]){
                objects[data.address] = current_objects[data.address];
            }else{
                var obj = graph.insertVertex(parent, null, node, 0, 0, 98, 89, styles[data.type]);
                objects[data.address] = obj;
            }

            if(data.type == 'ws'){
                graph.insertEdge(parent, null, '', objects[data.address], objects[data.network], 'rounded=0;fontSize=11;endArrow=none;endFill=0;jumpStyle=sharp;');    
            }
        }

        graph.refresh();

        var layout = new mxFastOrganicLayout(graph);
        layout.execute(parent);
    }
});

    function insert_velociraptor(result){
        var parent = graph.getDefaultParent();

        //var dataset = JSON.parse(result);
        
        var styles = {
            'windows': 'verticalLabelPosition=bottom;sketch=0;aspect=fixed;html=1;verticalAlign=top;strokeColor=none;align=center;outlineConnect=0;shape=mxgraph.citrix.desktop;',
            'network': '"sketch=0;html=1;aspect=fixed;strokeColor=none;shadow=0;align=center;verticalAlign=middle;fillColor=lightblue;shape=mxgraph.gcp2.network',
        }
        var networks = [];
        var objects = {};

            var cell = cells[id];
            if(cell.getAttribute('ClientId')){
                current_objects[cell.getAttribute('ClientId')] = cell;

                current_objects[cell.getAttribute('IPAddress')] = cell;
            }
        }

            }

            var data = JSON.parse(line);
            var ipv4_address = ipv4_check(data.IPAddress);
            var ipv4_subnet = ipv4_check(data.IPSubnet);
            var ipv4_gateway = ipv4_check(data.DefaultIPGateway);
            
            if(current_objects[data.ClientId]){
                var node = current_objects[data.ClientId];
            }else if(ipv4_address.length > 0 && current_objects[ipv4_address.join(', ')]){

            node.setAttribute('Architecture', data.Architecture);
            node.setAttribute('Platform', data.Platform);
            node.setAttribute('PlatformVersion', data.PlatformVersion);
            node.setAttribute('KernelVersion', data.KernelVersion);
            if(ipv4_address.length > 0){
                node.setAttribute('IPAddress', ipv4_address.join(', '));
            }
            if(ipv4_subnet.length > 0){
                node.setAttribute('IPSubnet', ipv4_subnet.join(', '));
            }
            if(ipv4_gateway.length > 0){
                node.setAttribute('DefaultIPGateway', ipv4_gateway.join(', '));
            }
            if(data.ListenPort && data.ListenPort.length > 0){
                node.setAttribute('Port', data.ListenPort.join(', '));
            }
            
            if(current_objects[data.ClientId]){
                objects[data.ClientId] = current_objects[data.ClientId];
            }else if(ipv4_address.length > 0 && current_objects[ipv4_address.join(', ')]){
                objects[ipv4_address.join(', ')] = current_objects[ipv4_address.join(', ')];
            }else{
                var obj = graph.insertVertex(parent, null, node, 0, 0, 98, 89, styles[data.OS]);
                objects[data.ClientId] = obj;
            }

            for(var i = 0; i < ipv4_address.length; i++){
                var network_address = get_network_address(ipv4_address[i], ipv4_subnet[i]);
                
                if(!current_objects[network_address] && !objects[network_address]){
                    var doc = mxUtils.createXmlDocument();
                    var node = doc.createElement('MyNode');
                    node.setAttribute('placeholders', 1);
                    node.setAttribute('label', '%IPAddress%');
                    node.setAttribute('IPAddress', network_address);
                    var obj = graph.insertVertex(parent, null, node, 0, 0, 98, 89, styles['network']);
                    objects[network_address] = obj;
                }

                graph.insertEdge(parent, null, '', objects[data.ClientId], objects[network_address], 'rounded=0;fontSize=11;endArrow=none;endFill=0;jumpStyle=sharp;');
            }
        }

        graph.refresh();

        var layout = new mxFastOrganicLayout(graph);
        layout.execute(parent);
    }

    function ipv4_check(lists){
        var result = [];

        if(Array.isArray(lists)){
            for(list of lists){
                if(list.includes('.')){
                    result.push(list);
                }
            }
        }else{
            result.push(lists);
        }
        
        return result;
    }

    function get_network_address(ip, subnet){
        var ip_bin = '';
        var subnet_bin = '';
        for(num of ip.split('.')){
            ip_bin += Number(num).toString(2).padStart(8, 0);
        }
        for(num of subnet.split('.')){
            subnet_bin += Number(num).toString(2).padStart(8, 0);
        }
        var mask = subnet_bin.match(/1/g).length;
        var network_bin = ip_bin.substr(0, mask).padEnd(32, 0);
        var network_address = parseInt(network_bin.substr(0, 8), 2) + '.' + parseInt(network_bin.substr(8, 8), 2) + '.' + parseInt(network_bin.substr(16, 8), 2) + '.' + parseInt(network_bin.substr(24, 8), 2);
        
        return network_address + '/' + mask;
    }
